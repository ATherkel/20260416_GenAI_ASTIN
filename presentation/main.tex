    \item Udsagn som \textquote{\emph{jeg lærer bedst ved at skrive ting ned}}\footnote{%
      \href{https://psycnet.apa.org/doiLanding?doi=10.1037\%2F0278-7393.4.6.592}

  \begin{block}{Opfordring:}
